query HideoutQuery {
  hideoutStations(gameMode: regular) {
    name
    id
    levels {
      level
      itemRequirements {
        count
        item {
          name
          shortName
          iconLink
          wikiLink
        }
      }
      skillRequirements {
        level
        name
      }
      traderRequirements {
        value
        requirementType
        trader {
          name
        }
      }
      stationLevelRequirements {
        level
        station {
          name
        }
      }
    }
  }
}
